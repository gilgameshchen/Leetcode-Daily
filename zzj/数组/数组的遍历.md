### 485.最大连续1的个数

```python
class Solution:
    def findMaxConsecutiveOnes(self, nums: List[int]) -> int:
        cnt = 0
        max = 0
        for num in nums:
            if num == 1:
                cnt = cnt + 1
            else:
                if cnt > max:
                    max = cnt
                cnt = 0
        if cnt > max:
            max = cnt
            cnt = 0
        return max
```

### 495.提莫攻击

```python
class Solution:
    def findPoisonedDuration(self, timeSeries: List[int], duration: int) -> int:
        cnt = 0
        l = len(timeSeries)
        for i in range(l-1):
            if timeSeries[i] + duration < timeSeries[i+1]:
                cnt = cnt + duration
            else:
                cnt = cnt + timeSeries[i+1] - timeSeries[i]
        cnt = cnt + duration
        return cnt
```

考虑前n-1次攻击会不会被下一次攻击重置时间，再加上最后一次不会被重置时间的攻击

### 414. 第三大的数

```python
class Solution:
    def thirdMax(self, nums: List[int]) -> int:
        q = []
        for num in nums:
            self.insert(q,num)
        if len(q) < 3:
            return q[0]
        else:
            return q[2]
    
    def insert(self, q :List[int], num :int):
        l = len(q)
        if l == 0:
            q.append(num)
            return
        if num not in q:
            for i in range(l):
                if num > q[i]:
                    q.insert(i,num)
                    break
            else:
                q.append(num)
        if len(q) == 4:
            q.pop()
```
